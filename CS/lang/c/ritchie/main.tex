    \unsect{Input / output}
    \begin{itemize}
        \item \textbf{stdio.h}: In C, input/output is mostly handled through the standard I/O library, declared in:
            \bigbreak \noindent 
            \begin{cppcode}
            #include <std.io>
            \end{cppcode}
        \item \textbf{Standard streams}: When a C program starts, it automatically has three standard streams:
            \bigbreak \noindent 
            \begin{cppcode}
            stdin
            stdout
            stderr
            \end{cppcode}
        \item \textbf{printf}: Printf prints formatted text to stdout.
            \bigbreak \noindent 
            \begin{cppcode}
            printf("%s %s", "Hello, ", "world!");
            \end{cppcode}
            \bigbreak \noindent 
            The placeholders are called format specifies. Common ones include
            \bigbreak \noindent 
            \begin{cppcode}
                %d      int
                %u      unsigned int
                %ld     long
                %lu     unsigned long
                %lld    long long
                %llu    unsigned long long

                %f      double
                %lf     double in scanf, but also accepted in printf
                %c      char
                %s      string / char array
                %p      pointer address
                %%      literal percent sign
            \end{cppcode}
            \bigbreak \noindent 
            For \%p, cast the pointer to void*. Note that the language itself does not make printf typesafe. Wrong format specifiers cause \textbf{undefined behavior}.
        \item \textbf{fprintf}: The fprintf function in C is used to write formatted data directly to an output stream, which is most commonly a file. It works identically to printf, except it requires a destination file pointer as its very first argument. 
            \bigbreak \noindent 
            \begin{cppcode}
            int fprintf(FILE *stream, const char *format, ...);
            \end{cppcode}
        \item \textbf{sprintf}: The sprintf function in C formats and stores a series of characters and values into a string buffer instead of printing them directly to the console. It is defined in the <stdio.h> standard library.
            \bigbreak \noindent 
            \begin{cppcode}
            int sprintf(char *str, const char *format, ...); 
            \end{cppcode}
            \bigbreak \noindent 
            Returns the total number of characters successfully written, excluding the null terminator (\textbackslash0). If an error occurs, it returns a negative value. 
        \item \textbf{snprintf}: 
            \bigbreak \noindent 
            \begin{cppcode}
            int snprintf ( char * s, size_t n, const char * format, ... );
            \end{cppcode}
            \bigbreak \noindent 
            Composes a string with the same text that would be printed if format was used on printf, but instead of being printed, the content is stored as a C string in the buffer pointed by s (taking n as the maximum buffer capacity to fill).
            \bigbreak \noindent 
            If the resulting string would be longer than n-1 characters, the remaining characters are discarded and not stored, but counted for the value returned by the function.
            \bigbreak \noindent 
            A terminating null character is automatically appended after the content written.
            \bigbreak \noindent 
            Returns the number of characters that would have been written if $n$ had been sufficiently large, not counting the terminating null character. If an encoding error occurs, a negative number is returned.
            \bigbreak \noindent 
            The key rule is \textbf{size} is the total buffer size, including space for the null terminator.
            \bigbreak \noindent 
            \begin{cppcode}
                char buf[10];
                snprintf(buf, sizeof buf, "hello");
            \end{cppcode}
            \bigbreak \noindent 
            sizeof buf is 10, so snprintf may write at most 9 visible characters plus the final '\textbackslash0'.
            \bigbreak \noindent 
            \begin{cppcode}
                char buf[6];
                snprintf(buf, sizeof buf, "hello");
            \end{cppcode}
            \bigbreak \noindent 
            This fits exactly. So, this is safe. 
            \bigbreak \noindent 
            \begin{cppcode}
                char buf[5];

                int n = snprintf(buf, sizeof buf, "hello");
            \end{cppcode}
            \bigbreak \noindent 
            snprintf writes
            \bigbreak \noindent 
            \begin{cppcode}
            'h' 'e' 'l' 'l' '\0'
            \end{cppcode}
            \bigbreak \noindent 
            in this case the return value is still $5$, because "hello" would have required 5 visible characters, not counting the null terminator. snprintf returns the number of characters that would have been written, excluding the null terminator.
            \bigbreak \noindent 
            \begin{cppcode}
                char buf[5];

                int n = snprintf(buf, sizeof buf, "hello");

                if (n < 0) {
                    // encoding or formatting error
                } else if ((size_t)n >= sizeof buf) {
                    // output was truncated
                } else {
                    // output fit
                }
            \end{cppcode}
            \bigbreak \noindent 
            You can call snprintf once to find out how much space is needed:
            \bigbreak \noindent 
            \begin{cppcode}
                int n = snprintf(NULL, 0, "value = %d", 123);
            \end{cppcode}
        \item \textbf{scanf}: scanf reads formatted input from stdin.
            \bigbreak \noindent 
            \begin{cppcode}
            int x;
            scanf("%d", &x)
            \end{cppcode}
            \bigbreak \noindent 
            Scanf needs the address because it has to modify the variable. Scanf returns the number of successful conversions. You should usually check the return value
            \bigbreak \noindent 
            \begin{cppcode}
            int x;
            if (scanf("%d", &x) != 1) {
                fprintf(stderr, "Invalid integer\n");
            }
            \end{cppcode}
        \item \textbf{puts}: Writes a string to the console
            \bigbreak \noindent 
            \begin{cppcode}
            puts("Hello, world!");
            \end{cppcode}
        \item \textbf{getchar}: reads one character from stdin
            \bigbreak \noindent 
            \begin{cppcode}
            int c;
            c = getchar();
            \end{cppcode}
            \bigbreak \noindent 
            Notice that getchar returns int, not char. That is because it may return EOF, which is a special value indicating end-of-file or error.
            \bigbreak \noindent 
            \begin{cppcode}
            int c;
            while ((c=getchar()) != EOF) putchar(c);
            \end{cppcode}
        \item \textbf{putchar}: Writes one character to stdout
            \bigbreak \noindent 
            \begin{cppcode}
            putchar('A');
            \end{cppcode}
        \item \textbf{fgets for line input}:  Avoid
            \bigbreak \noindent 
            \begin{cppcode}
                char name[32];
                scanf("%s", name);
            \end{cppcode}
            \bigbreak \noindent 
            If the user types more than 31 characters, you can overflow the buffer. A safer approach is fgets:
            \bigbreak \noindent 
            \begin{cppcode}
                char name[32];

                printf("Enter your name: ");

                if (fgets(name, sizeof name, stdin) == NULL) {
                    fprintf(stderr, "Input error\n");
                    return 1;
                }

                printf("Hello, %s", name);
            \end{cppcode}
            \bigbreak \noindent 
            Fgets has signature
            \bigbreak \noindent 
            \begin{cppcode}
            char* fgets( char* str, int count, FILE* stream );
            \end{cppcode}
            \bigbreak \noindent 
            and returns \textbf{str} on success, null pointer on failure.
        \item \textbf{sscanf}: The sscanf function in C extracts and parses formatted data directly from a character string rather than reading from standard keyboard input (stdin). It is defined in the <stdio.h> library and stands for "String Scan Formatted". 
            \bigbreak \noindent 
            \begin{cppcode}
            int sscanf(const char *str, const char *format, ...);
            \end{cppcode}
            \bigbreak \noindent 
            \begin{cppcode}
                const char *info = "ID:4023 Alice 28";
                int id;
                char name[20];
                int age;

                // Parse data by explicitly matching the literal "ID:" prefix
                int items = sscanf(info, "ID:%d %19s %d", &id, name, &age);
            \end{cppcode}
            \begin{itemize}
                \item \textbf{\%[set]:} A scanset. For instance, %[A-Za-z] reads only alphabetic characters.
                \item \textbf{\%[^set]:} An inverted scanset. For instance, %[^,] reads everything up until it hits a comma.
                \item \textbf{* Modifier:} Suppresses assignment. For example, %*d matches and skips an entire integer without storing it.
            \end{itemize}


    \end{itemize}

    \unsect{Working with files}
    \begin{itemize}
        \item \textbf{Header}: File I /O is mainly done through the same header
            \bigbreak \noindent 
            \begin{cppcode}
            #include <stdio.h>
            \end{cppcode}
        \item \textbf{FILE type}: The central type is
            \bigbreak \noindent 
            \begin{cppcode}
            FILE*
            \end{cppcode}
            \bigbreak \noindent 
            A FILE* is a handle to an open file stream. You do not manipulate the file directly. You open it, get a FILE *, use functions such as fgetc, fgets, fprintf, fread, and finally close it with fclose.
        \item \textbf{fopen}: 
            \bigbreak \noindent 
            \begin{cppcode}
            FILE *fopen(const char *restrict filename, const char *restrict mode);
            \end{cppcode}
            \bigbreak \noindent 
            Upon successful completion fopen(), fdopen(), and freopen() return a FILE pointer.  Otherwise, NULL is returned and errno is set to indicate the error.
            \bigbreak \noindent 
            The mode string tells C whether you want to read, write, append, or work in binary mode. 
            \bigbreak \noindent 
            \begin{center}
                \begin{tabularx}{\textwidth}{@{}lX@{}}
                    \toprule
                    \textbf{Mode}	&\textbf{Meaning} \\
                    \midrule
                    "r"	    &Open existing file for reading\\[2ex]
                    "w"	    &Open file for writing; destroys old contents\\[2ex]
                    "a"	    &Open file for appending\\[2ex]
                    "r+"	&Open existing file for reading and writing\\[2ex]
                    "w+"	&Open for reading and writing; destroys old contents\\[2ex]
                    "a+"	&Open for reading and appending\\[2ex]
                    "rb"	&Read binary\\[2ex]
                    "wb"	&Write binary\\[2ex]
                    "ab"	&Append binary \\
                    \bottomrule
                \end{tabularx}
            \end{center}
            \bigbreak \noindent 
            The primary difference between r+ and a+ modes in C file handling lies in how they handle file creation and where the data is written. While both allow you to read and write without erasing (truncating) the existing data, they behave very differently under the hood. 
            \bigbreak \noindent 
            \begin{center}
                \begin{tabularx}{\textwidth}{@{}XXX@{}}
                    \toprule
                    \textbf{Feature}& \textbf{r+ (Read / Update)}&	\textbf{a+ (Append / Update)} \\
                    \midrule
                    File Must Exist?	&Yes (returns NULL if missing)	&No (creates a new file if missing)\\[2ex]
                    Initial Pointer Position	&Beginning of the file	&End of the file\\[2ex]
                    Overwrites Content?	&Yes (overwrites at current cursor)	&No (always forces writes to EOF)\\[2ex]
                    Effect of fseek() on Writes	&Moves the cursor for the next write	&Ignored (writes always go to the end) \\ 
                    \bottomrule
                \end{tabularx}
            \end{center}
            \bigbreak \noindent 
            \begin{cppcode}
                #include <stdio.h>

                int main(void) {
                    FILE *fp = fopen("data.txt", "r");

                    if (fp == NULL) {
                        perror("fopen");
                        return 1;
                    }

                    fclose(fp);
                    return 0;
                }
            \end{cppcode}
        \item \textbf{fclose}: 
            \bigbreak \noindent 
            \begin{cppcode}
            int fclose(FILE* stream);
            \end{cppcode}
            \bigbreak \noindent 
            fclose closes an open file stream. It also flushes any buffered output before closing the file.
            \bigbreak \noindent 
            You should call fclose on every file you successfully opened with fopen.
            \bigbreak \noindent 
            Upon successful completion, 0 is returned.  Otherwise, EOF is returned and errno is set to indicate the error.
            \bigbreak \noindent 
            EOF is a negative integer constant defined in <stdio.h>.
            \bigbreak \noindent 
            \begin{cppcode}
            #include <stdio.h>

            int main(void) {
                FILE *fp = fopen("output.txt", "w");

                if (fp == NULL) {
                    perror("fopen");
                    return 1;
                }

                fprintf(fp, "Hello file\n");

                if (fclose(fp) == EOF) {
                    perror("fclose");
                    return 1;
                }

                return 0;
            }
            \end{cppcode}
        \item \textbf{The restrict keyword}: restrict is a type qualifier used on pointers to promise the compiler that the pointer is the only means to access its underlying memory object
            \bigbreak \noindent 
            When you qualify a pointer with restrict, you guarantee that for the lifetime of that pointer, no other pointer or variable inside the same scope will read or write to that specific chunk of memory.
        \item \textbf{fprintf}: 
            \bigbreak \noindent 
            \begin{cppcode}
            int fprintf(FILE *restrict stream, const char *restrict format, ...);
            \end{cppcode}
            \bigbreak \noindent 
            fprintf works like printf, but it writes to a file stream instead of standard output.
            \bigbreak \noindent 
            Returns the number of characters written on success.
            \bigbreak \noindent 
            Returns a negative value on failure.
            \bigbreak \noindent 
            \begin{cppcode}
                #include <stdio.h>

                int main(void) {
                    FILE *fp = fopen("numbers.txt", "w");

                    if (fp == NULL) {
                        perror("fopen");
                        return 1;
                    }

                    int x = 10;
                    double y = 3.14;

                    fprintf(fp, "x = %d\n", x);
                    fprintf(fp, "y = %.2f\n", y);

                    fclose(fp);
                    return 0;
                }
            \end{cppcode}
        \item \textbf{fputs}: 
            \bigbreak \noindent 
            \begin{cppcode}
                int fputs(const char *restrict s, FILE *restrict stream);
            \end{cppcode}
            \bigbreak \noindent 
            fputs writes a string to a file. Unlike puts, it does not automatically add a newline.
            \bigbreak \noindent 
            Returns a non-negative value on success. Returns EOF on failure.
        \item \textbf{fputc}
            \bigbreak \noindent 
            \begin{cppcode}
            int fputc(int c, FILE *stream);
            \end{cppcode}
            \bigbreak \noindent 
            fputc writes a single character to a file. The character is passed as an int, but it usually represents a char.
            \bigbreak \noindent 
            Returns the character written on success. Returns EOF on failure.
        \item \textbf{fgetc}: 
            \bigbreak \noindent 
            \begin{cppcode}
            int fgetc(FILE *stream);
            \end{cppcode}
            \bigbreak \noindent 
            fgetc reads one character from a file.
            \bigbreak \noindent 
            It returns an int, not a char, because it needs to represent every possible character plus the special value EOF.
            \bigbreak \noindent 
            Returns the character read on success. Returns EOF if the end of the file is reached or if an error occurs.
            \bigbreak \noindent 
            \begin{cppcode}
                #include <stdio.h>

                int main(void) {
                    FILE *fp = fopen("data.txt", "r");

                    if (fp == NULL) {
                        perror("fopen");
                        return 1;
                    }

                    int ch;

                    while ((ch = fgetc(fp)) != EOF) {
                        putchar(ch);
                    }

                    fclose(fp);
                    return 0;
                }
            \end{cppcode}
        \item \textbf{fgets}: 
            \begin{cppcode}
                char *fgets(char *restrict s, int n, FILE *restrict stream);
            \end{cppcode}
            \bigbreak \noindent 
            fgets reads a line, or part of a line, from a file into a character buffer.
            \bigbreak \noindent 
            It reads at most $n - 1$ characters and then adds the null terminator '\textbackslash 0'.
            \bigbreak \noindent 
            If it sees a newline before the limit, it includes the newline in the buffer.
            \bigbreak \noindent 
            \begin{cppcode}
                #include <stdio.h>

                int main(void) {
                    FILE *fp = fopen("data.txt", "r");

                    if (fp == NULL) {
                        perror("fopen");
                        return 1;
                    }

                    char line[256];

                    while (fgets(line, sizeof line, fp) != NULL) {
                        printf("%s", line);
                    }

                    fclose(fp);
                    return 0;
                }
            \end{cppcode}









    \end{itemize}
