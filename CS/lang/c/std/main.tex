    \unsect{Stdio}
    \subsection{Output}
    \begin{itemize}
        \item \bigbreak \noindent 
            \begin{cppcode}
                int printf(const char *restrict format, ...);
            \end{cppcode}
            \bigbreak \noindent 
            Upon successful return, these functions return the number of bytes printed (excluding the null byte used to end output to strings).
            \bigbreak \noindent 
            On error, a negative value is returned, and errno is set to indicate the error.
        \item \bigbreak \noindent 
            \begin{cppcode}
                int fprintf(FILE *stream, const char *format, ...);
            \end{cppcode}
            \bigbreak \noindent 
            Upon successful return, these functions return the number of bytes printed (excluding the null byte used to end output to strings).
            \bigbreak \noindent 
            On error, a negative value is returned, and errno is set to indicate the error.
        \item \bigbreak \noindent 
            \begin{cppcode}
                int sprintf(char *str, const char *format, ...); 
            \end{cppcode}
            \bigbreak \noindent 
            Returns the total number of characters successfully written, excluding the null terminator (\textbackslash0). 
            \bigbreak \noindent 
            If an error occurs, it returns a negative value and errno is set.
        \item \bigbreak \noindent 
            \begin{cppcode}
                int snprintf ( char * s, size_t n, const char * format, ... );
            \end{cppcode}
            \bigbreak \noindent 
            Returns the number of characters that would have been written if $n$ had been sufficiently large, not counting the terminating null character. 
            \bigbreak \noindent 
            If an encoding error occurs, a negative number is returned and errno is set.
        \item \bigbreak \noindent \begin{cppcode}
                int fputc(int c, FILE *stream);
                int putc(int c, FILE *stream);
                int putchar(int c);
            \end{cppcode}
            \bigbreak \noindent 
            Return the character written as an unsigned char cast to an int or EOF on error.
        \item \bigbreak \noindent 
            \begin{cppcode}
                int fputs(const char *restrict s, FILE *restrict stream);
                int puts(const char *s);
            \end{cppcode}
            \bigbreak \noindent 
            Returns a nonnegative number on success, or EOF on error.



    \end{itemize}

    \subsection{Input}
    \begin{itemize}
        \item \bigbreak \noindent 
            \begin{cppcode}
                int scanf(const char *restrict format, ...);
                int fscanf(FILE *restrict stream, const char *restrict format, ...);
            \end{cppcode}
            \bigbreak \noindent 
            On success, these functions return the number of input items successfully matched and assigned; this can be fewer than provided for, or even zero, in the event of an early matching failure.
            \bigbreak \noindent 
            The value EOF is returned if the end of input is reached before either the first successful conversion or a matching failure occurs.  EOF is also returned if a read error occurs, in which case the error indicator for the stream is set, and errno is set to indicate the error.
        \item \bigbreak \noindent 
            \begin{cppcode}
                int getchar(void);
                int fgetc(FILE *stream);
            \end{cppcode}
            \bigbreak \noindent 
            Upon successful completion, fgetc() shall return the next byte from the input stream pointed to by stream.  If the end-of-file indicator for the stream is set, or if the stream is at end-of- file, the end-of-file indicator for the stream shall be set and fgetc() shall return EOF. If a read error occurs, the error indicator for the stream shall be set, fgetc() shall return EOF, and shall set errno to indicate the error.
            \bigbreak \noindent 
            The return behavior for getchar is the same as fgetc.
        \item \bigbreak \noindent 
            \begin{cppcode}
             int sscanf(const char *restrict str, const char *restrict format, ...);
             \end{cppcode}
             \bigbreak \noindent 
             On success, these functions return the number of input items successfully matched and assigned; this can be fewer than provided for, or even zero, in the event of an early matching failure.
             \bigbreak \noindent 
             The value EOF is returned if the end of input is reached before either the first successful conversion or a matching failure occurs.
        \item \bigbreak \noindent 
            \begin{cppcode}
               char *fgets(char *restrict s, int n, FILE *restrict stream);
            \end{cppcode}
            \bigbreak \noindent 
            Upon successful completion, fgets() shall return s.  If the stream is at end-of-file, the end-of-file indicator for the stream shall be set and fgets() shall return a null pointer.  If a read error occurs, the error indicator for the stream shall be set, fgets() shall return a null pointer, and shall set errno to indicate the error.

    \end{itemize}

    \subsection{Binary I/O}
    \begin{itemize}
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
                size_t fwrite(const void *restrict data_ptr, size_t element_size, size_t n_elements, FILE *restrict stream); 
            \end{cppcode}
            \bigbreak \noindent 
             The fwrite() function shall return the number of elements successfully written, which may be less than nitems if a write error is encountered. If size or nitems is 0, fwrite() shall return 0 and the state of the stream remains unchanged. Otherwise, if a write error occurs, the error indicator for the stream shall be set, and errno shall be set to indicate the error.
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
            size_t fread(void ptr[restrict size * n], size_t size, size_t n, FILE *restrict stream);
            \end{cppcode}
            \bigbreak \noindent 
            The function fread() reads n items of data, each size bytes long, from the stream pointed to by stream, storing them at the location given by ptr.
            \bigbreak \noindent 
            On success, fread() and fwrite() return the number of items read or written.  This number equals the number of bytes transferred only when size is 1.  If an error occurs, or the end of the file is reached, the return value is a short item count (or zero).

    \end{itemize}

            \bigbreak \noindent 
            Returns current position on success, -1L on failure.
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
            void rewind(FILE* stream);
            \end{cppcode}
            \bigbreak \noindent 
            reset the file position indicator in a stream

            \bigbreak \noindent 
            \begin{cppcode}

            \end{cppcode}
            \bigbreak \noindent 
            Returns nothing.
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
                int atexit(void (*func)(void));
            \end{cppcode}
            \bigbreak \noindent 
            Returns zero on success, nonzero on failure.
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
            void _Exit(int status);
            \end{cppcode}
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
            int system(const char* string);
            \end{cppcode}
            \bigbreak \noindent 
            Returns implementation-defined status information.
            \bigbreak \noindent 
            The return value is 
            \begin{itemize}
                \item If command is NULL, then a nonzero value if a shell is available, or 0 if no shell is available.
                \item  If a child process could not be created, or its status could not be retrieved, the return value is -1 and errno is set to indicate the error.
                \item If a shell could not be executed in the child process, then the return value is as though the child shell terminated by calling \_exit(2) with the status 127.
                \item  If all system calls succeed, then the return value is the termination status of the child shell used to execute command. (The termination status of a shell is the termination status of the last command it executes.)
            \end{itemize}
        \item 
            \bigbreak \noindent 
            \begin{cppcode}
            char* getenv(const char* name);
            \end{cppcode}
            \bigbreak \noindent 
            Returns a pointer to the value string, or NULL if the variable does not exist.



    \end{itemize}
