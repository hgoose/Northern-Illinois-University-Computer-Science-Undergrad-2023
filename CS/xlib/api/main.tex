            \bigbreak \noindent

                #define BitmapNoMemory      3

        \item \textbf{Context manager status values}: Used by Xlib context-manager functions.
            \bigbreak \noindent 
            \begin{cppcode}

                #define XCNOENT     2
            \end{cppcode}
        \item \textbf{Locale / converter status values}: Used by locale and text conversion setup functions.
            \bigbreak \noindent 
            \begin{cppcode}
                #define XNoMemory              -1

                #define XConverterNotFound     -3

        \item \textbf{Grab status values}: Used by grab functions such as XGrabPointer and XGrabKeyboard.

    \unsect{Display functions}
    \begin{itemize}
        \item \textbf{Structures:} 
            \begin{cppcode}
                typedef struct {
                    int depth;
                    int bits_per_pixel;
                    int scanline_pad;
                } XPixmapFormatValues;
            \end{cppcode}

    \item \textbf{Functions}:
        \begin{cppcode}
            Display *XOpenDisplay(char* display_name)
        \end{cppcode}
        \begin{itemize}
            \item \textbf{Description}: Opens a connection to the X server and returns a \texttt{Display*} used for nearly all later Xlib calls.
            \item \textbf{Possible errors}:
                \begin{itemize}
                    \item \textit{None}: This function does not generate X protocol errors.
                    \item Failure to open the display is indicated by a \texttt{NULL} return value.
                \end{itemize}
        \end{itemize}
        \textbf{Note:} Passing NULL tells X to use the DISPLAY environment variable.

        \begin{cppcode}
            AllPlanes

        \begin{itemize}
            \item \textbf{Description}: Returns a bit mask with all plane bits set; commonly used when copying or querying all drawable planes.
