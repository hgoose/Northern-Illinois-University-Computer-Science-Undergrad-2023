    \begin{mdframed}
        1. Solve the following ODE
        \begin{align*}
            y^{\prime\prime} + 3y^{\prime} + 4y = 3x + 2
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We first find $y_{c}$ by examining the roots of the characteristic equation
    \begin{align*}
        r^{2} + 3r + 4 &= r^{2} + 3r + \left(\frac{3}{2}\right)^{2} + 4 - \left(\frac{3}{2}\right)^{2} = \left(r+\frac{3}{2}\right)^{2}= 0 \\
        \implies \left(r+\frac{3}{2}\right)^{2} &= -\frac{7}{4} \implies r = -\frac{3}{2}\pm \frac{\sqrt{7}}{2}i
    .\end{align*}
    Thus,
    \begin{align*}
        y_{c} = c_{1}e^{-\frac{3}{2}x}\cos{\left(\frac{\sqrt{7}}{2}x\right)} + c_{2}e^{-\frac{3}{2}x}\sin{\left(\frac{\sqrt{7}}{2}x\right)}
    .\end{align*}
    Then, we find $y_{p}$ by assuming its form and solving for its coefficients, we have
    \begin{align*}
        y_{p} = Ax + B, \quad y^{\prime}_{p} = A, \quad y^{\prime\prime}_{p} = 0
    .\end{align*}
    So,
    \begin{align*}
        0 + 3A + 4\left(Ax+B\right) = 3A+4Ax + 4B = 3x + 2
    .\end{align*}
    Now, we can compare coefficients,
    \begin{align*}
        4A &= 3 \implies A = \frac{3}{4} \\
        3A + 4B &= 2 \implies 3\left(\frac{3}{4}\right) + 4B = 2 \implies B = \frac{1}{4}\left(2-\frac{9}{4}\right) = -\frac{1}{16}
    .\end{align*}
    With these coefficients, we conclude that
    \begin{align*}
        y = y_{c} + y_{p} = c_{1}e^{-\frac{3}{2}x}\cos{\left(\frac{\sqrt{7}}{2}x\right)} + c_{2}e^{-\frac{3}{2}x}\sin{\left(\frac{\sqrt{7}}{2}x\right)} + \frac{3}{4}x - \frac{1}{16}
    .\end{align*}
    

    \bigbreak \noindent 
    \begin{mdframed}
        2. Solve the following ODE
        \begin{align*}
            y^{\prime\prime} - 4y  = 2e^{3x}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    The characteristic is 
    \begin{align*}
        r^{2}-4 = (r-2)(r+2) = 0
    .\end{align*}
    So, $r=\pm 2$, and 
    \begin{align*}
        y_{c} = c_{1}e^{2x} + c_{2}e^{-2x}
    .\end{align*}
    The particular solution $y_{p}$ has the form
    \begin{align*}
        y_{p} = Ae^{3x}
    ,\end{align*}
    so $y_{p}^{\prime} = 3Ae^{3x}$, $y^{\prime\prime}_{p} = 9Ae^{3x}$, and 
    \begin{align*}
        9Ae^{3x}-4Ae^{3x} = 2e^{3x}\implies 9A-4A = 2 \implies 5A=2 \implies A = \frac{2}{5}
    .\end{align*}
    Thus, the solution is
    \begin{align*}
        c_{1}e^{2x} + c_{2}e^{-2x} + \frac{2}{5}e^{3x}
    .\end{align*}


    \bigbreak \noindent 
    \begin{mdframed}
        3. Solve the following ODE
        \begin{align*}
            y^{\prime\prime} - 4y = 2e^{2x}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have the same complementary solution as question 2,
    \begin{align*}
        y_{c} = c_{1}e^{2x} + c_{2}e^{-2x}
    ,\end{align*}
    except now, since $f(x) = 2e^{2x}$ is linearly dependent with $e^{2x}$ in the fundamental solution set of the homogeneous part, we let 
    \begin{align*}
        y_{p} = Axe^{2x}
    .\end{align*}
    So, $y_{p}^{\prime} = Ae^{2x} + 2Axe^{2x}$, $y^{\prime\prime}_{p} = 2Ae^{2x} + 2Ae^{2x} + 4Axe^{2x} = 4Ae^{2x} + 4Axe^{2x} $, and 
    \begin{align*}
        4Ae^{2x} + 4Axe^{2x} - 4Axe^{2x} = 4Ae^{2x} = 2e^{2x}        
    .\end{align*}
    Thus,
    \begin{align*}
        4A = 2 \implies A = \frac{2}{4} = \frac{1}{2}
    .\end{align*}
    With this, we have the solution
    \begin{align*}
        y = y_{c} + y_{p} = c_{1}e^{2x} + c_{2}e^{-2x} + \frac{1}{2}xe^{2x}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        4. Solve the following ODE
        \begin{align*}
            3y^{\prime\prime} + y^{\prime} - 2y = 2\cos{\left(x\right)}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have the characteristic equation
    \begin{align*}
        3r^{2} + r - 2 = 3\left(r^{2}+\frac{1}{3}r - \frac{2}{3}\right) = 3\left(r+1\right)\left(r-\frac{2}{3}\right) = 0
    .\end{align*}
    Thus, $r=-1,\; r=\frac{2}{3}$, and
    \begin{align*}
        y_{c} = c_{1}e^{-x} + c_{2}e^{\frac{2}{3}x}
    .\end{align*}
    The particular solution $y_{p}$ has the form
    \begin{align*}
        y_{p} = A\cos{\left(x\right)} + B\sin{\left(x\right)}
    ,\end{align*}
    with $y_{p}^{\prime} = -A\sin{\left(x\right)} + B\cos{\left(x\right)} $, $y^{\prime\prime}_{p} = -A\cos{\left(x\right)} -B\sin{\left(x\right)}$, and 
    \begin{align*}
        3\left(-A\cos{\left(x\right)} - B\sin{\left(x\right)}\right) -A\sin{\left(x\right)} + B\cos{\left(x\right)} - 2\left(A\cos{\left(x\right)} + B\sin{\left(x\right)}\right) &= 2\cos{\left(x\right)} \\
        \implies (-3A-2A+B)\cos{\left(x\right)} + (-3B-A-2B)\sin{\left(x\right)} &= 2\cos{\left(x\right)}
    .\end{align*}
    Thus,
    \begin{align*}
        -A-5B &= 0 \implies A = -5B \\
        -5A+B &= 2  \implies -5(-5B) + B = 2 \implies 26B = 2 \implies B = \frac{1}{13}
    .\end{align*}
    So $A = -\frac{5}{13}$ and
    \begin{align*}
        y = y_{c} + y_{p} = c_{1}e^{-x} + c_{2}e^{\frac{2}{3}x} -\frac{5}{13}\cos{\left(x\right)} + \frac{1}{13}\sin{\left(x\right)}
    .\end{align*}
