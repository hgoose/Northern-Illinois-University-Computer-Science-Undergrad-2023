    Nate Warner \ \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad  MATH 336 \quad  \quad \quad \quad \quad \quad \quad \quad \quad \ \ \quad \quad Spring 2026
    \begin{center}
        \textbf{Problem set 9 - Due: Monday, March 23}
    \end{center}
    \bigbreak \noindent 
    Solve the following ODEs using variation of parameters, given the complementary function $y_{c}$


    \begin{mdframed}
        1. 
        \begin{align*}
            x^{2}y^{\prime\prime} -4xy^{\prime} + 6y = x^{3}, \quad y_{c} = c_{1}x^{2} + c_{2}x^{3}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        y^{\prime\prime} - \frac{4}{x}y^{\prime} + \frac{6}{x^{2}}y = x
    .\end{align*}
    So, with
    \begin{align*}
        W = x^{2}(3x^{2}) - x^{3}(2x) = 3x^{4} - 2x^{4} = x^{4}
    ,\end{align*}
    we have the particular solution
    \begin{align*}
        y_{p} &= -x^{2}\int \frac{x^{3}x}{x^{4}}\; dx + x^{3}\int \frac{x^{2}x}{x^{4}}\; dx \\
              &=-x^{2}x + x^{3}\ln{\left\lvert x \right\rvert} = x^{3}\ln{\left\lvert x \right\rvert} - x^{3}
    .\end{align*}
    Thus, the solution is
    \begin{align*}
        y = c_{1}x^{2} + c_{2}x^{3} + x^{3}\ln{\left\lvert x \right\rvert} - x^{3} = c_{1}x^{2} + c_{2}x^{3} + x^{3}\ln{\left\lvert x \right\rvert}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        2. 
        \begin{align*}
            4x^{2}y^{\prime\prime} - 4xy^{\prime} + 3y = 8x^{\frac{4}{3}}, \quad y_{c} = c_{1}x^{\frac{1}{2}} + c_{2}x^{\frac{3}{2}}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        y^{\prime\prime} - \frac{1}{x}y^{\prime} + \frac{3}{4x^{2}}y = 8x^{\frac{4}{3}-2} = 8x^{-\frac{2}{3}}
    .\end{align*}
    With 
    \begin{align*}
        W = x^{\frac{1}{2}}\left(\frac{3}{2}x^{\frac{1}{2}}\right) - x^{\frac{3}{2}}\left(\frac{1}{2}x^{-\frac{1}{2}}\right) = \frac{3}{2}x - \frac{1}{2}x = x
    ,\end{align*}
    we get the particular solution
    \begin{align*}
        y_{p} &= -x^{\frac{1}{2}}\int \frac{x^{\frac{3}{2}}8x^{-\frac{2}{3}}}{x}\; dx + x^{\frac{3}{2}}\int \frac{x^{\frac{1}{2}}8x^{-\frac{2}{3}}}{x}\; dx \\
        &= -x^{\frac{1}{2}}\int 8x^{-\frac{1}{6}}\;dx +x^{\frac{3}{2}}\int 8x^{-\frac{7}{6}}\;dx \\
        &= -x^{\frac{1}{2}}\left(\frac{48}{5}\right)x^{\frac{5}{6}} +x^{\frac{3}{2}}(-48)x^{-\frac{1}{6}} \\
        &=-\frac{48}{5}x^{\frac{8}{6}} - 48x^{\frac{8}{6}} = -\frac{6}{5}(48)x^{\frac{8}{6}} = -\frac{288}{5}x^{\frac{4}{3}}
    .\end{align*}
    Thus, the solution is
    \begin{align*}
        y = c_{1}x^{\frac{1}{2}} + c_{2}x^{\frac{3}{2}} -\frac{288}{5}x^{\frac{4}{3}}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        3. 
        \begin{align*}
            x^{2}y^{\prime\prime} - 2xy^{\prime} + 2y = x^{\frac{9}{2}}, \quad y_{c} = c_{1}x + c_{2}x^{2}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        y^{\prime\prime} - \frac{2}{x}y^{\prime}+\frac{2}{x^{2}}y = x^{\frac{5}{2}}
    .\end{align*}
    Now, since
    \begin{align*}
        W = x(2x) - x^{2} = 2x^{2} -x^{2} = x^{2}
    ,\end{align*}
    we have the particular solution
    \begin{align*}
        y_{p} &= -x\int \frac{x^{2}x^{\frac{5}{2}}}{x^{2}}\; dx + x^{2}\int \frac{xx^{\frac{5}{2}}}{x^{2}}\; dx \\
        &=-x\int x^{\frac{5}{2}}\; dx + x^{2}\int x^{\frac{3}{2}}\; dx \\
        &= -\frac{2}{7}xx^{\frac{7}{2}} + \frac{2}{5}x^{2}x^{\frac{5}{2}} = -\frac{2}{7}x^{\frac{9}{2}} + \frac{2}{5}x^{\frac{9}{2}} = \frac{4}{35}x^{\frac{9}{2}}
    .\end{align*}
    Thus, the solution is
    \begin{align*}
        y = c_{1}x + c_{2}x^{2} + \frac{4}{35}x^{\frac{9}{2}}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        4. 
        \begin{align*}
            x^{2}y^{\prime\prime} - xy^{\prime} - 3y = x^{\frac{3}{2}}, \quad y_{c} = c_{1}\frac{1}{x} + c_{2}x^{3}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        y^{\prime\prime} - \frac{1}{x}y^{\prime} - \frac{3}{x^{2}}y = x^{-\frac{1}{2}}
    .\end{align*}
    So, with
    \begin{align*}
         W = \frac{1}{x}(3x^{2}) - \left(-\frac{1}{x^{2}}x^{3}\right) = 3x+x = 4x
    ,\end{align*}
    we have 
    \begin{align*}
        y_{p} &= -\frac{1}{x}\int \frac{x^{3}x^{-\frac{1}{2}}}{4x}\; dx + x^{3}\int \frac{\frac{1}{x}x^{-\frac{1}{2}}}{4x}\; dx \\
        &=-\frac{1}{x}\int \frac{1}{4}x^{\frac{3}{2}}\; dx + x^{3}\int \frac{1}{4}x^{-\frac{5}{2}}\; dx \\
        &=-\frac{1}{4x}\left(\frac{2}{5}\right)x^{\frac{5}{2}} + \frac{x^{3}}{4}\left(-\frac{2}{3}\right)x^{-\frac{3}{2}} \\
        &=-\frac{1}{10}x^{\frac{3}{2}} -\frac{1}{6}x^{\frac{3}{2}} = -\frac{4}{15}x^{\frac{3}{2}}
    .\end{align*}
    So,
    \begin{align*}
        y = c_{1}\frac{1}{x} + c_{2}x^{3} -\frac{4}{15}x^{\frac{3}{2}}
    .\end{align*}
