    \begin{mdframed}
        1. Determine whether the functions is linearly independent on the real line.
        \begin{align*}
            f(x) = \pi, \quad g(x) = \sin^{2}{\left(x\right)} + \cos^{2}{\left(x\right)}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    \begin{remark}
       The \textit{Wronskian} is the determinant of the matrix formed by $n$ functions and their derivatives,
       \begin{align*}
           W(y_{1}, \dots, y_{n}) = \det\begin{pmatrix} 
               y_{1} & y_{2} & \cdots & y_{n} \\
               y^{\prime}_{1} & y^{\prime}_{2} & \cdots & y^{\prime}_{n} \\
               \vdots & \vdots & \ddots &  \vdots \\
               y^{(n-1)}_{1} & y^{(n-1)}_{2} & \cdots & y^{(n-1)}_{n} 
           \end{pmatrix}
       \end{align*}
       If $W = 0$, the set formed by $y_{1}, \dots, y_{n}$ is linearly dependent, and linearly independent otherwise. $\endpf$
    \end{remark}
    \bigbreak \noindent 
    The Wronskian $W(\pi, 1)$ is given by
    \begin{align*}
        \det\begin{pmatrix} \pi & 1 \\ 0 & 0 \end{pmatrix} = \pi(0) - 1(0) = 0
    .\end{align*}
    Thus, $f$ and $g$ are linearly dependent.

    \begin{mdframed}
        2. Determine whether the functions is linearly independent on the real line.
        \begin{align*}
            f(x) = 17, \quad g(x) = \cos^{2}{\left(x\right)}, \quad h(x) = \cos{\left(2x\right)}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    The Wronskian for these functions is 
    \begin{align*}
        W(17, \cos^{2}{\left(x\right)}, \cos{\left(2x\right)}) &= \det\begin{pmatrix} 
            17 & \cos^{2}{\left(x\right)} & \cos{\left(2x\right)} \\
            0 & -2\cos{\left(x\right)}\sin{\left(x\right)} & -2\sin{\left(2x\right)} \\
            0 & 2\sin^{2}{\left(x\right)} - 2\cos^{2}{\left(x\right)} & -4\cos{\left(2x\right)}
            \end{pmatrix} = \det\begin{pmatrix} 
                17 &  \cos^{2}{\left(x\right)} & \cos{\left(2x\right)} \\
                0 & -\sin{\left(2x\right)} & -2\sin{\left(2x\right)} \\
                0 & -2\cos{\left(2x\right)} & -4\cos{\left(2x\right)}
            \end{pmatrix} \\
                                                               &= 17\left(4\sin{\left(2x\right)}\cos{\left(2x\right)} - 4\sin{\left(2x\right)}\cos{\left(2x\right)}\right) = 17(0) = 0
    .\end{align*}
    Thus, these functions are linearly dependent.

    \begin{mdframed}
        3. Verify that $y_{1} = e^{3x}$ and $y_{2} = e^{-3x}$ are solutions of the differential equation $y^{\prime\prime} - 9y = 0$. Then find a particular solution of the form $y = c_{1}y_{1} + c_{2}y_{2}$ that satisfies the initial conditions $y(0) = -1, y^{\prime}(0) = 15$.
    \end{mdframed}
    \bigbreak \noindent 
    We verify that these functions are solutions to the differential equation $y^{\prime\prime} - 9y = 0$ by first solving the differential equation (just kidding). Instead, we compute the proposed solutions derivatives and then plug them into the ODE, we have
    \begin{align*}
        y_{1}^{\prime} = 3e^{3x},\; y_{1}^{\prime\prime} = 9e^{3x}, \quad y_{2}^{\prime} = -3e^{-3x}, \; y_{2}^{\prime\prime} = 9e^{-3x}
    .\end{align*}
    Regarding $y_{1}$, we get
    \begin{align*}
       9e^{3x} -9(e^{3x}) = 0 \implies 0 = 0 
    .\end{align*}
    Next,
    \begin{align*}
        9e^{-3x} - 9(e^{-3x}) = 0 \implies 0 = 0 
    .\end{align*}
    Thus, the solutions are verified. Notice that we have a homogeneous linear ODE with constant coefficients, so we use the superposition principle to form the general solution
    \begin{align*}
        y = c_{1}e^{3x} + c_{2}e^{-3x}
    .\end{align*}
    With the initial conditions $y(0) = -1$ and $y^{\prime}(0) = 15$, we can determine the constants $c_{1}, c_{2}$, 
    \begin{align*}
        y(0) = -1 = c_{1}e^{0} + c_{2}e^{0} \implies c_{1} = -1-c_{2}
    .\end{align*}
    Next, since
    \begin{align*}
        y^{\prime} = 3c_{1}e^{3x} - 3c_{2}e^{-3x}
    ,\end{align*}
    we have
    \begin{align*}
        y^{\prime}(0) &= 15 = 3c_{1}e^{0} - 3c_{2}e^{0} = 3(-1-c_{2})-3c_{2} = -3-3c_{2}-3c_{2} = -3-6c_{2} \\
        \implies c_{2} &= -3
    .\end{align*}
    It then follows that $c_{1} = -1 + 3 = 2$. So, the particular solution is
    \begin{align*}
        y = 2e^{3x} -3e^{-3x}
    .\end{align*}

    \begin{mdframed}
        4. Find the general solution of the following differential equation
        \begin{align*}
            2y^{\prime\prime} - 7y^{\prime} + 3y = 0
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have the characteristic equation
    \begin{align*}
        2r^{2}- 7r + 3  = 2\left(r^{2}-\frac{7}{2}r + \frac{3}{2}\right) = 2(r-3)\left(r-\frac{1}{2}\right)= 0
    .\end{align*}
    Thus, $r=3$ and $r=\frac{1}{2}$, which implies that the general solution is
    \begin{align*}
        y = c_{1}e^{3x} + c_{2}e^{\frac{1}{2}x}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        5. Find the general solution of the following differential equation
        \begin{align*}
            y^{\prime\prime} + 6y^{\prime} + 9y = 0
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    The characteristic equation is 
    \begin{align*}
        r^{2} + 6r + 9 = (r+3)^{2} = 0
    .\end{align*}
    Thus, $r=-3$ with multiplicity $2$, which implies the general solution is
    \begin{align*}
        y = c_{1}e^{-3x}  +c_{2}xe^{-3x}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        6. Find the general solution of the following differential equation
        \begin{align*}
            y^{\prime\prime} + 8y^{\prime} + 25y = 0
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    The characteristic equation is 
    \begin{align*}
        r^{2} + 8r + 25 =0
    .\end{align*}
    Observe that the discriminant is 
    \begin{align*}
        D = 8^{2}-4(25) < 0
    ,\end{align*}
    so we will have two complex roots, those roots are precisely
    \begin{align*}
        \frac{-b\pm\sqrt{D}}{2a} = \frac{-8\pm \sqrt{-36}}{2} = -\frac{8}{2} \pm \frac{6i}{2} = -4\pm 3i
    .\end{align*}
    Thus, the general solution is
    \begin{align*}
        y = c_{1}e^{-4x}\cos{\left(3x\right)} + c_{2}e^{-4x}\sin{\left(3x\right)}
    .\end{align*}
