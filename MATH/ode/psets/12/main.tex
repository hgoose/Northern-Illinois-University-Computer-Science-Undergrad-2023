    \begin{mdframed}
        1. A body with mass $m = \frac{1}{2}$ kilogram (kg) is attached to the end of a spring stretched 2 meters (m) by a force of 100 newtons (N). It is set in motion with initial position $x_{0} = 1(m)$ and initial velocity $v_{0} = −5(m/s)$. (Note that these initial conditions indicate that the body is displaced to the right and is moving to the left at time $t = 0$.) Derive the equation of motion and find the position function of the body.
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        100 &= k(2) \implies k = 50, \\
    \end{align*}
    with no damper and no external force. Thus, the ODE is 
    \begin{align*}
        \frac{1}{2}x^{\prime\prime} + 50x = 0,\quad x(0) = 1,\; x^{\prime}(0) = -5 \frac{m}{s}
    .\end{align*}
    We have the C.E
    \begin{align*}
        \frac{1}{2}r^{2} + 50 = 0 = r^{2} + 100
    .\end{align*}
    So,
    \begin{align*}
        r &= \pm 10i
    .\end{align*}
    Thus,
    \begin{align*}
        x(t) &= c_{1}\cos{\left(10t\right)} + c_{2}\sin{\left(10t\right)}, \\
        x^{\prime}(t) &= -10c_{1}\sin{\left(10t\right)} +10c_{2}\cos{\left(10t\right)}
    .\end{align*}
    With the initial conditions
    \begin{align*}
        1 &= c_{1}\cos{\left(0\right)} + c_{2}\sin{\left(0\right)} = c_{1}, \\
        -5 &= -10(1)\sin{\left(0\right)} + 10c_{2}\cos{\left(0\right)} = 10c_{2}
    .\end{align*}
    Thus, $c_{1} = 1,\; c_{2} = -\frac{1}{2}$, and the equation is therefore
    \begin{align*}
        x(t) = \cos{\left(10t\right)} -\frac{1}{2} \sin{\left(10t\right)}
    .\end{align*}

    \begin{mdframed}
        2. An object stretches a spring 6 inches in equilibrium. Find its displacement for $t > 0$ if it is initially displaced 18 inches above equilibrium and given a downward velocity of 3ft/s
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        mg = kx \implies k = \frac{mg}{x} = \frac{mg}{6\; \text{in}} = \frac{mg}{0.5\; \text{ft}} = 2mg
    .\end{align*}
    With 
    \begin{align*}
        mx^{\prime\prime} + kx = 0
    ,\end{align*}
    we get 
    \begin{align*}
        x^{\prime\prime} + \frac{k}{m}x = 0 = x^{\prime\prime} + \frac{2mg}{m}x
    .\end{align*}
    Thus,
    \begin{align*}
        x^{\prime\prime} + 2gx = 0
    .\end{align*}
    Using $g = 32\; \text{ft}/s^{2}$, 
    \begin{align*}
        x^{\prime\prime} + 64x = 0
    ,\end{align*}
    With initial conditions $x_{0} = 18\; \text{in} = \frac{3}{2}\; \text{ft} $, $x^{\prime}(0) = -3\; \text{ft}/s $. The characteristic equation is
    \begin{align*}
        r^{2} + 64 = 0
    .\end{align*}
    Thus, $r = \pm 8i $, and
    \begin{align*}
        x(t) &= c_{1}\cos{\left(8t\right)} + c_{2}\sin{\left(8t\right)}, \\
        x^{\prime}(t) &= -8c_{1}\sin{\left(8t\right)} + 8c_{2}\cos{\left(8t\right)}
    .\end{align*}
    Thus,
    \begin{align*}
        \frac{3}{2} &= c_{1}\cos{\left(0\right)} + c_{2}\sin{\left(0\right)} = c_{1}
    ,\end{align*}
    and
    \begin{align*}
        -3 &= -8\left(\frac{3}{2}\right)\sin{\left(8t\right)} + 8c_{2}\cos{\left(8t\right)} \\
           &=-12\sin{\left(0\right)} + 8c_{2}\cos{\left(0\right)} = 8c_{2}
    .\end{align*}
    Thus, $c_{1} = \frac{3}{2} $ and $c_{2} = -\frac{3}{8} $. With these constants, we have 
    \begin{align*}
        x(t) = \frac{3}{2}\cos{\left(8t\right)} - \frac{3}{8}\sin{\left(8t\right)}
    .\end{align*}
