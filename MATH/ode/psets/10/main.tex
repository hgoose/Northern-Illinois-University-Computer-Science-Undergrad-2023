        1. Find the inverse Laplace transform of
        \begin{align*}
            F(s) = \frac{1}{s^{2} - 9}
        .\end{align*}
    \end{mdframed}
    \begin{align*}
        \mathcal{L}^{-1}\left\{\frac{1}{s^{2}-  9}\right\} = \frac{1}{3}\mathcal{L}^{-1}\left\{\frac{3}{s^{2}-3^{2}}\right\} =\frac{1}{3}\sinh{\left(3t\right)}
    .\end{align*}
    \bigbreak \noindent 
    \begin{mdframed}
       2. Find the inverse Laplace transform of
        \begin{align*}
            F(s) = \frac{2}{s^{2}(s^{2} + 36)}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    Using partial fractions,
    \begin{align*}
        \frac{2}{s^{2}(s^{2}+36)} = \frac{A}{s^{2}} + \frac{Bs+C}{s^{2}+36}
    .\end{align*}
    Thus,
    \begin{align*}
        2 = A(s^{2} + 36) + (Bs + C)s^{2} = As^{2} + 36A + Bs^{3} + Cs^{2} = (B)s^{3} + (A+C)s^{2} + 36A
    .\end{align*}
    Hence,
    \begin{align*}
        B &= 0, \\
        A+C &= 0, \\
        36A &= 2 
    .\end{align*}
    So,
    \begin{align*}
        A &= \frac{1}{18}, \quad C = -\frac{1}{18}, \quad B = 0
    .\end{align*}
    Therefore,
    \begin{align*}
        \mathcal{L}^{-1}\left\{\frac{2}{s^{2}(s^{2}+36)}\right\} &= \mathcal{L}^{-1}\left\{\frac{1}{18s^{2}}\right\} + \mathcal{L}^{-1}\left\{-\frac{1}{18(s^{2}+36)}\right\} \\
        &=\frac{1}{18}\mathcal{L}^{-1}\left\{\frac{1}{s^{2}}\right\} - \frac{1}{18\cdot 6}\mathcal{L}^{-1}\left\{\frac{6}{s^{2}+6^{2}}\right\} \\
        &=\frac{1}{18}t - \frac{1}{18\cdot 6}\sin{\left(6t\right)} = \frac{1}{18}t - \frac{1}{108}\sin{\left(6t\right)}
    .\end{align*}

    \begin{mdframed}
        3. Solve the following initial value problem using Laplace transform
        \begin{align*}
            x^{\prime\prime} - x^{\prime} - 6x = 0, \quad x(0) = 2,\; x^{\prime}(0) = -1
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        \mathcal{L}\left\{x^{\prime\prime} - x^{\prime} - 6x\right\} = \mathcal{L}\left\{0\right\} = 0 \\
        \implies s^{2}X(s) - sx(0) - x^{\prime}(0) - \left(sX(s) - x(0)\right) - 6X(s) = 0 \\
        \implies s^{2}X(s) - 2s + 1 - sX(s) + 2 -6X(s) = 0  \\
        \implies (s^{2}-s-6)X(s) - 2s + 3 = 0 \\
        \implies (s+2)(s-3)X(s) -2s + 3 = 0 \\
        \implies X(s) = \frac{2s}{(s+2)(s-3)} - \frac{3}{(s+2)(s-3)}
    .\end{align*}
    Then, by the inverse Laplace,
    \begin{align*}
        x(t) = \mathcal{L}^{-1}\left\{\frac{2s}{(s+2)(s-3)}\right\} - \mathcal{L}^{-1}\left\{\frac{3}{(s+2)(s-3)}\right\}
    .\end{align*}
    Using partial fractions,
    \begin{align*}
        \frac{2s}{(s+2)(s-3)} = \frac{A}{s+2} + \frac{B}{s-3}
    .\end{align*}
    So,
    \begin{align*}
        2s  =A(s-3) + B(s+2)
    .\end{align*}
    Let $s = -2$, then 
    \begin{align*}
        2(-2) = -5A = -4 \implies A = \frac{4}{5}
    .\end{align*}
    Let $s=3$, then
    \begin{align*}
        2(3) = 5B \implies B = \frac{6}{5}
    .\end{align*}
    So,
    \begin{align*}
        \mathcal{L}^{-1}\left\{\frac{2s}{(s+2)(s-3)}\right\} &= \mathcal{L}^{-1}\left\{\frac{4}{5(s+2)}\right\} + \mathcal{L}^{-1}\left\{\frac{6}{5(s-3)}\right\} \\
                                                             &= \frac{4}{5}\mathcal{L}^{-1}\left\{\frac{1}{s+2}\right\} + \frac{6}{5}\mathcal{L}^{-1}\left\{\frac{1}{s-3}\right\} \\
                                                             &= \frac{4}{5}e^{-2t} + \frac{6}{5}e^{3t}
    .\end{align*}
    Next, 
    \begin{align*}
        \frac{3}{(s+2)(s-3)} = \frac{A}{s+2} + \frac{B}{s-3} \implies 3 = A(s-3) + B(s+2)
    .\end{align*}
    Then,
    \begin{align*}
        3 = -5A \implies A = -\frac{3}{5}
    ,\end{align*}
    and 
    \begin{align*}
        3 = 5B \implies B = \frac{3}{5}
    .\end{align*}
    Hence,
    \begin{align*}
        \mathcal{L}^{-1}\left\{\frac{3}{(s+2)(s-3)}\right\} &= -\frac{3}{5}\mathcal{L}^{-1}\left\{\frac{1}{s+2}\right\} + \frac{3}{5}\mathcal{L}^{-1}\left\{\frac{1}{s-3}\right\} \\
        &=-\frac{3}{5}e^{2t} + \frac{3}{5}e^{-3t}
    .\end{align*}
    The full solution is then
    \begin{align*}
        x(t) &= \frac{4}{5}e^{-2t} + \frac{6}{5}e^{3t}  - \left(-\frac{3}{5}e^{-2t} + \frac{3}{5}e^{3t}\right) \\
        &= \frac{4}{5}e^{-2t} + \frac{6}{5}e^{3t}  + \frac{3}{5}e^{-2t} - \frac{3}{5}e^{3t} \\
        &= \left(\frac{4}{5} + \frac{3}{5}\right)e^{-2t} + \left(\frac{6}{5}-\frac{3}{5}\right)e^{-3t} \\
        &= \frac{7}{5}e^{-2t} + \frac{3}{5}e^{-3t}
    .\end{align*}

    \begin{mdframed}
        4. Solve the following initial value problem using Laplace transform
        \begin{align*}
            x^{\prime\prime} + 4x = \sin{\left(3t\right)}, \quad x(0) = x^{\prime}(0) = 0
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        \mathcal{L}\left\{x^{\prime\prime} + 4x\right\} &= \mathcal{L}\left\{\sin{\left(3t\right)}\right\} \\
        \implies s^{2}X(s) - sx(0) - x^{\prime}(0) + 4X(s) &= \frac{3}{s^{2} + 3^{2}} \\
        \implies (s^{2} + 4)X(s) &= \frac{3}{s^{2} + 9} \\
        \implies X(s) &= \frac{3}{(s^{2} + 9)(s^{2} + 4)}
    .\end{align*}
    By partial fractions
    \begin{align*}
        X(s) &= \frac{3}{(s^{2}+9)(s^{2}+4)} \\[4pt]
             &= \frac{A}{s^{2}+9} + \frac{B}{s^{2}+4}.
    \end{align*}
    Thus,
    \begin{align*}
        3 &= A(s^{2}+4) + B(s^{2}+9) \\
          &= As^{2} + 4A + Bs^{2} + 9B \\
          &= (A+B)s^{2} + (4A+9B).
    \end{align*}
    Hence,
    \begin{align*}
        A+B &= 0, \\
        4A+9B &= 3.
    \end{align*}
    From \(A+B=0\), we have \(A=-B\), so
    \begin{align*}
        4(-B)+9B &= 3 \\
        5B &= 3 \\
        B &= \frac{3}{5}.
    \end{align*}
    Thus,
    \begin{align*}
        A &= -\frac{3}{5}.
    \end{align*}
    So
    \begin{align*}
        X(s)
    &= -\frac{3}{5}\frac{1}{s^{2}+9}
    + \frac{3}{5}\frac{1}{s^{2}+4}.
    \end{align*}
    By inverse Laplace, 
    \begin{align*}
        x(t) &= -\frac{3}{5}\mathcal{L}^{-1}\left\{\frac{1}{s^{2} + 3^{2}}\right\} + \frac{3}{5}\mathcal{L}^{-1}\left\{\frac{1}{s^{2}+2^{2}}\right\} \\
        &= -\frac{3}{15}\mathcal{L}^{-1}\left\{\frac{3}{s^{2}+3^{2}}\right\} + \frac{3}{10}\mathcal{L}^{-1}\left\{\frac{2}{s^{2}+2^{2}}\right\} \\
        &=-\frac{1}{5}\sin{\left(3t\right)} + \frac{3}{10}\sin{\left(2t\right)}
    .\end{align*}
