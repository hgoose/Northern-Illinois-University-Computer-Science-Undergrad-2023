    \begin{mdframed}
        1. Use the second translation theorem
        \begin{align*}
            \mathcal{L}^{-1}\left\{e^{-as}F(s)\right\} = u(t-a)f(t-a)
        \end{align*}
        to find the inverse of 
        \begin{align*}
            F(s) = \frac{e^{-3s}}{s^{2}}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        F(s) = e^{-3s}\frac{1}{s^{2}}
    .\end{align*}
    Thus,
    \begin{align*}
        f(t) = \mathcal{L}^{-1}\left\{e^{-3s}\frac{1}{s^{2}}\right\}
    .\end{align*}
    Let $G(s) = \frac{1}{s^{2}}$, then 
    \begin{align*}
        g(t) = \mathcal{L}^{-1}\left\{\frac{1}{s^{2}}\right\} = t
    .\end{align*}
    So, $g(t-3)  = t-3$, and 
    \begin{align*}
        \mathcal{L}^{-1}\left\{e^{-3s}\frac{1}{s^{2}}\right\} = u(t-3)(t-3)
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        2. Use the second translation theorem
        \begin{align*}
            \mathcal{L}^{-1}\left\{e^{-as}F(s)\right\} = u(t-a)f(t-a)
        \end{align*}
        to find the inverse of 
        \begin{align*}
            F(s) = e^{-s}\frac{1}{s+2}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    Let $G(s) = \frac{1}{s+2}$, then
    \begin{align*}
        g(t) = \mathcal{L}^{-1}\left\{\frac{1}{s+2}\right\} = e^{-2t}
    .\end{align*}
    Since $a = 1$, $g(t-1) = e^{-2(t-1)}$. Therefore, 
    \begin{align*}
       \mathcal{L}^{-1}\left\{e^{-s}\frac{1}{s+2}\right\}  = u(t-1)e^{-2(t-1)}
    .\end{align*}

    \pagebreak 
    \begin{mdframed}
        3. Use the second translation theorem
        \begin{align*}
            \mathcal{L}^{-1}\left\{e^{-as}F(s)\right\} = u(t-a)f(t-a)
        \end{align*}
        to find the inverse of 
        \begin{align*}
            F(s) = e^{-s}\frac{s}{s^{2} + \pi^{2}}
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    Let $G(s) = \frac{s}{s^{2}+\pi^{2}}$, then
    \begin{align*}
        g(t) = \mathcal{L}^{-1}\left\{\frac{s}{s^{2} + \pi^{2}}\right\} = \cos{\left(\pi t \right)}
    .\end{align*}
    Then, since $a = 1$, 
    \begin{align*}
        f(t) = u(t-1)\cos{\left(\pi(t-1)\right)}
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        3. Use Laplace transform method to solve the ODE
        \begin{align*}
            x^{\prime\prime} + 4x = 
            \begin{cases}
                \cos{\left(2t\right)}     &\quad 0  \leq t < 2\pi \\
                0 &\quad t \geq 2\pi
            \end{cases}, x(0) = x^{\prime}(0) = 0
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        \begin{cases}
            \cos{\left(2t\right)}     &\quad 0  \leq t < 2\pi \\
            0 &\quad t \geq 2\pi
        \end{cases}
        =
        \begin{cases}
            h     &\quad 0  \leq t < 2\pi \\
            h &\quad t \geq 2\pi
        \end{cases}
        +
        \begin{cases}
            0     &\quad 0  \leq t < 2\pi \\
            f(t-2\pi) &\quad t \geq 2\pi
        \end{cases}
    .\end{align*}
    Thus,
    \begin{align*}
        \cos{\left(2t\right)}  &= h, \\
        0 = \cos{\left(2t\right)} + f(t-2\pi) \implies f(t-2\pi) &= -\cos{\left(2t\right)}
    .\end{align*}
    So,
    \begin{align*}
        \begin{cases}
            \cos{\left(2t\right)}     &\quad 0  \leq t < 2\pi \\
            0 &\quad t \geq 2\pi
        \end{cases}
        =
        \cos{\left(2t\right)} - \cos{\left(2t\right)}u(t-2\pi)
    .\end{align*}
    Then,
    \begin{align*}
        \mathcal{L}\left\{\cos{\left(2t\right)}\right\} - \mathcal{L}\left\{\cos{\left(2t\right)}u(t-2\pi)\right\} &= \frac{s}{s^{2}+4} - e^{-2\pi s}\frac{s}{s^{2} + 4} \\
                                                                                                                   &= (1-e^{-2\pi s})\frac{s}{s^{2}+4}
    .\end{align*}
    So, we have
    \begin{align*}
        \mathcal{L}\left\{x^{\prime\prime} + 4x\right\} &= (1-e^{-2\pi s})\frac{s}{s^{2}+4} \\
        \implies s^{2}X(s) -sx(0) - x^{\prime}(0) + 4X(s) &= (1-e^{-2\pi s})\frac{s}{s^{2}+4} \\
        \implies (s^{2} + 4)X(s) &= (1-e^{-2\pi s})\frac{s}{s^{2}+4} \\
        \implies X(s) &= (1-e^{-2\pi s})\frac{s}{(s^{2}+4)^{2}} 
    .\end{align*}
    Thus,
    \begin{align*}
        X(s) &= \frac{s}{(s^{2}+4)^{2}} - e^{-2\pi s}\frac{s}{(s^{2}+4)^{2}}
    ,\end{align*}
    so
    \begin{align*}
        x(t) &= \mathcal{L}^{-1}\left\{\frac{s}{(s^{2}+4)^{2}}\right\} - \mathcal{L}^{-1}\left\{e^{-2\pi s}\frac{s}{(s^{2}+4)^{2}}\right\}
    .\end{align*}
    With 
    \begin{align*}
        \mathcal{L}^{-1}\left\{\frac{s}{(s^{2}+4)^{2}}\right\} = \frac{1}{4}t\sin{\left(2t\right)}
    ,\end{align*}
    we let $G(s) = \frac{s}{(s^{2}+4)^{2}}$, so
    \begin{align*}
        g(t) = \frac{1}{4}t\sin{\left(2t\right)}
    .\end{align*}
    With $a = 2\pi$, $g(t-2\pi) = \frac{1}{4}(t-2\pi)\sin{\left(2(t-2\pi)\right)} $, so
    \begin{align*}
        x(t) &= \frac{1}{4}t\sin{\left(2t\right)} - u(t-2\pi)\frac{1}{4}(t-2\pi)\sin{\left(2(t-2\pi)\right)}
    .\end{align*}
