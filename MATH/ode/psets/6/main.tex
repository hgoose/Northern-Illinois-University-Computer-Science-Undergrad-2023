    \begin{mdframed}
        1. The population of a town grows at a rate proportional to the population present at time $t$. The initial population of 500 increases by 10\% in 10 years. What will be the population in 20 years?
    \end{mdframed}
    \bigbreak \noindent 
    \begin{remark}
       We have 
       \begin{align*}
           \frac{dP}{dt} = kP 
       ,\end{align*}
       which implies that
       \begin{align*}
           P(t) = Ce^{kt}
       .\end{align*}
       If $P(0) = P_{0}$, then $C = P_{0}$, and
       \begin{align*}
           P(t) = P_{0}e^{kt}
       .\end{align*}
       $\endpf$
    \end{remark}
    \bigbreak \noindent 
    We are given $P_{0} =500$, and $P(10) = 550$, so 
    \begin{align*}
        550 = 500e^{10k} \implies e^{10k} = \frac{55}{50} = \frac{11}{10} \implies 10k = \ln{\left(\frac{11}{10}\right)} \implies k = \frac{1}{10}\ln{\left(\frac{11}{10}\right)}
    .\end{align*}
    Thus,
    \begin{align*}
        P(t) = 500e^{\frac{t}{10}\ln{\left(\frac{11}{10}\right)}} = 500\left(\frac{11}{10}\right)^{\frac{t}{10}}
    .\end{align*}
    With this,
    \begin{align*}
        P(20) = 500\left(\frac{11}{10}\right)^{\frac{20}{10}} = 500\left(\frac{11}{10}\right)^{2}  = 605
    .\end{align*}

    

    \bigbreak \noindent 
    \begin{mdframed}
        2. The population of bacteria in a culture grows at a rate proportional to the number of bacteria present at time t. After 3 hours it is observed that 400 bacteria are present. After 10 hours 2000 bacteria are present. What was the initial number of bacteria?
    \end{mdframed}
    \bigbreak \noindent 
    We have
    \begin{align*}
        P(3) = 400 = P_{0}e^{3k}, \quad P(10) = 2000 = P_{0}e^{10k}
    .\end{align*}
    So,
    \begin{align*}
        P_{0} = \frac{400}{e^{3k}} = \frac{2000}{e^{10k}}
    .\end{align*}
    Thus,
    \begin{align*}
        400e^{10k} &= 2000e^{3k} \implies \frac{2000}{400} = 5 = e^{7k} \\
        \implies 7k &= \ln{\left(5\right)} \implies k = \frac{1}{7}\ln{\left(5\right)}
    .\end{align*}
    Using $P(3)$ and our value for $k$, we can see that
    \begin{align*}
       400 = P_{0}e^{\frac{3}{7}\ln{\left(5\right)}}  = P_{0}5^{\frac{3}{7}} 
    .\end{align*}
    Therefore, $P_{0}  = \frac{400}{5^{\frac{3}{7}}} \approx 200.6788$.


    \bigbreak \noindent 
    \begin{mdframed}
        3. The population of a community is known to increase at a rate proportional to the number of people present at time $t$. If an initial population $P_{0}$ has doubled in 5 years, how long will it take to triple?
        \begin{enumerate}[label=(\alph*)]
            \item Suppose it is known that the population of the community is 10,000 after 3 years. What was the initial population $P_{0}$?
            \item What will be the population in 10 years?
        \end{enumerate}
    \end{mdframed}
    \bigbreak \noindent 
    If $P(5) = 2P_{0}$, then
    \begin{align*}
        P_{0}e^{5k} = 2P_{0} \implies e^{5k} = 2 \implies k = \frac{1}{5}\ln{\left(2\right)}
    .\end{align*}
    So, to find the time it takes for the population to triple, we solve for $t$ in the equation
    \begin{align*}
       P(t) = P_{0}e^{kt}  = P_{0}e^{\frac{t}{5}\ln{\left(2\right)}} = P_{0}\left(2\right)^{\frac{t}{5}}= 3P_{0} 
    .\end{align*}
    So, $t$ is precisely
    \begin{align*}
        2^{\frac{t}{5}}   = 3 \implies \frac{t}{5}\ln{\left(2\right)} = \ln{\left(3\right)} \implies t = \frac{5\ln{\left(3\right)}}{\ln{\left(2\right)}}
    .\end{align*}
    If it is known that the population of the community is 10,000 after just 3 years, then the initial population can be easily found. Observe
    \begin{align*}
        P(3) = 10000 = P_{0}2^{\frac{3}{5}}
    .\end{align*}
    Therefore, $P_{0} = \frac{10000}{2^{\frac{3}{5}}} \approx 6597.5396$
    \bigbreak \noindent 
    Finally, we can find $P(10)$,
    \begin{align*}
        P(10) = \frac{10000}{2^{\frac{3}{5}}}2^{2} = 10^{4} \cdot 2^{\frac{7}{5}} \approx 26390.1582
    .\end{align*}

    \bigbreak \noindent 
    \begin{mdframed}
        4. Suppose that at time $t = 0$, half of a “logistic” population of 100,000 persons have heard a certain rumor, and the number of those who have heard it is then increasing at the rate of 1000 persons per day. How long will it take for this rumor to spread to 80\% of the population? (Hint: Find the value of $k$ by substituting $P(0)$ and $P^{\prime}(0)$ in the logistic equation 
        \begin{align*}
            \frac{dP}{dt} = kP(M - P)
        .\end{align*}
    \end{mdframed}
    \bigbreak \noindent
    \begin{remark}
                    Let $M$ be the maximum capacity of the population, then
            \begin{align*}
                \frac{dP}{dt} = kP(M-P)
            .\end{align*}
            To solve this differential equation, we can use partial fractions, by separability, we have
            \begin{align*}
                \int \frac{1}{P(M-P)}\; dP = \int k\; dt
            .\end{align*}
            Using partial fractions, we get
            \begin{align*}
                \frac{1}{P(M-P)} = \frac{A}{p} + \frac{B}{M-P}
            .\end{align*}
            Then,
            \begin{align*}
                1 = A(M-P) + BP = AM + P(B-A)
            .\end{align*}
            So,
            \begin{align*}
                AM &= 1 \implies A = \frac{1}{M}, \\
                B-A &= 0 \implies B = \frac{1}{M}
            .\end{align*}
            From this, the LHS integral becomes
            \begin{align*}
                \int \frac{1}{P(M-P)}\; dP = \int \frac{1}{MP}\; dP + \int \frac{1}{M(M-P)}\; dP
            .\end{align*}
            Thus,
            \begin{align*}
                \frac{1}{m}\left(\ln{\left\lvert P \right\rvert} - \ln{\left\lvert M-P \right\rvert}\right) = kt + C
            .\end{align*}
            After some simple algebra, we obtain
            \begin{align*}
                P(t) = \frac{MCe^{kMt}}{1+Ce^{kMt}}
            .\end{align*}
            Now, if we let $P_{0} = P(0)$, then 
            \begin{align*}
                P_{0} = \frac{MC}{1+C} \implies C = \frac{P_{0}}{M-P_{0}}
            .\end{align*}
            With this, 
            \begin{align*}
                P(t) = \frac{M\left(\frac{P_{0}}{M-P_{0}}\right)e^{kMt}}{1+\left(\frac{P_{0}}{M-P_{0}}\right)e^{kMt}}
            .\end{align*}
            Then, after some even simpler algebra, we obtain 
            \begin{align*}
                P(t) = \frac{MP_{0}e^{kMt}}{P_{0}e^{kMt} + (M-P_{0})} 
            .\end{align*}
            $\endpf $
    \end{remark}
    \bigbreak \noindent 
    We have that $M=10^{5}$, $P_{0} = 10^{5}/2 = 5 \cdot 10^{4} $, and $P^{\prime}(0) = 10^{3}$. We require a time $t$ such that $P(t) = 0.8 \cdot 10^{5} = 8 \cdot 10^{4} $. First, we find $k$,
    \begin{align*}
        P^{\prime}(0) = 1000 = kP_{0}(M-P_{0}) = k(5)(10^{4})(5)(10^{4}) = k(5\cdot 10^{4})^{2}
    .\end{align*}
    Thus,  
    \begin{align*}
        k = \frac{10^{3}}{(5\cdot 10^{4})^{2}} = \frac{10^{3}}{25 \cdot 10^{8}} = \frac{1}{25 \cdot 10^{5}}
    .\end{align*}
    Note that 
    \begin{align*}
        kM = \frac{10^{5}}{25 \cdot 10^{5}} = \frac{1}{25}
    .\end{align*}
    Thus, 
    \begin{align*}
        e^{kMt} = e^{\frac{t}{25}}
    .\end{align*}
    So,
    \begin{align*}
        P(t) &= 8 \cdot 10^{4} = \frac{10^{5}(5\cdot 10^{4})e^{\frac{t}{25}}}{(5\cdot 10^{4})e^{\frac{t}{25}} + (5\cdot 10^{4})} = \frac{10^{5}e^{\frac{t}{25}}}{e^{\frac{t}{25}} +1} \\
        \implies \frac{10^{5}}{8\cdot 10^{4}} &= \frac{5}{4} = \frac{e^{\frac{t}{25}} + 1}{e^{\frac{t}{25}}} = 1 + \frac{1}{e^{\frac{t}{25}}} \\
        \implies \frac{1}{4} &= \frac{1}{e^{\frac{t}{25}}} \implies e^{\frac{t}{25}} = 4 \implies \frac{t}{25} = \ln{\left(4\right)} \\
        \implies t &= 25\ln{\left(4\right)} \approx 34.6574
    .\end{align*}
    Thus, the rumor will spread to $80\%$ of the population in approximately $34.66$ days.
